> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a CRM tab at `site/crm.html` showing a searchable, sortable table of all Customer - PLG accounts with live HubSpot data and placeholder columns for future data sources.

**Architecture:** Extend `scripts/fetch-hubspot.js` to batch-fetch the most-recently-created contact per company from HubSpot and write it into `hubspot.json`. Build `site/crm.html` as a static page that loads the JSON at runtime and renders a client-side searchable/sortable table. Add a CRM nav link to existing pages.

**Tech Stack:** Node.js (fetch script), HubSpot CRM API v3, plain HTML/CSS/vanilla JS

- [ ] **Step 1: Add `fetchContactIdsForCompanies` function**

```js
async function fetchContactIdsForCompanies(companies) {
  // Returns Map<companyId, contactId[]>
  const result = new Map();
  const CHUNK = 100;

  for (let i = 0; i < companies.length; i += CHUNK) {
    const chunk = companies.slice(i, i + CHUNK);
    const inputs = chunk.map(c => ({ id: c.hubspot_id, limit: 100 }));

    const data = await rpc('hubspot_batch_read_associations', {
      fromObjectType: 'companies',
      toObjectType: 'contacts',
      inputs,
    });

    const results = data?.result?.results ?? data?.results ?? [];
    for (const item of results) {
      const ids = (item.to ?? []).map(t => t.id);
      result.set(item.from.id, ids);
    }
  }

  return result;
}
```

- [ ] **Step 2: Add `fetchContacts` function**

```js
async function fetchContacts(contactIds) {
  // Returns Map<contactId, { name, email, createdate, _id }>
  const result = new Map();
  const CHUNK = 100;

  for (let i = 0; i < contactIds.length; i += CHUNK) {
    const chunk = contactIds.slice(i, i + CHUNK);
    const data = await rpc('hubspot_batch_read_contacts', {
      inputs: chunk.map(id => ({ id })),
      properties: ['firstname', 'lastname', 'email', 'createdate'],
    });

    const contacts = data?.result?.results ?? data?.results ?? [];
    for (const c of contacts) {
      result.set(c.id, {
        name:       [c.properties.firstname, c.properties.lastname].filter(Boolean).join(' ') || null,
        email:      c.properties.email ?? null,
        createdate: c.properties.createdate ?? null,
        _id:        c.id,
      });
    }
  }

  return result;
}
```

- [ ] **Step 3: Add `pickPrimaryContact` helper**

```js
function pickPrimaryContact(contactIds, contactMap) {
  const contacts = contactIds.map(id => contactMap.get(id)).filter(Boolean);
  if (contacts.length === 0) return null;

  contacts.sort((a, b) => {
    const da = a.createdate ? new Date(a.createdate).getTime() : 0;
    const db = b.createdate ? new Date(b.createdate).getTime() : 0;
    if (db !== da) return db - da;                               // most recent first
    return parseInt(a._id, 10) - parseInt(b._id, 10);           // tiebreak: lowest numeric ID
  });

  const { name, email } = contacts[0];
  return (name || email) ? { name, email } : null;
}
```

**Files:**
- Create: `site/crm.html`

- [ ] **Step 1: Create `site/crm.html`**

```html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>CRM — StackOne CS</title>
  <link rel="stylesheet" href="style.css">
</head>
<body>

  <header class="site-header">
    <div class="site-header-inner">
      <a href="index.html" class="site-logo">
        <span class="logo-dot"></span>
        CS Analytics
      </a>
      <nav class="site-nav">
        <a href="dashboard.html" class="site-nav-link">PLG Motion</a>
        <a href="crm.html" class="site-nav-link site-nav-link--active">CRM</a>
        <a href="index.html" class="site-nav-link">Support Digests</a>
      </nav>
    </div>
  </header>

  <main class="page-main">

    <div class="crm-toolbar">
      <input
        id="crm-search"
        class="crm-search"
        type="search"
        placeholder="Search customers…"
        autocomplete="off"
      >
      <span id="crm-count" class="crm-count"></span>
    </div>

    <div id="crm-table-wrap" class="crm-table-wrap">
      <table class="crm-table">
        <thead>
          <tr>
            <th id="th-name" aria-sort="ascending">Name</th>
            <th id="th-contact">Primary contact (HubSpot)</th>
            <th>Account (Pylon)</th>
            <th>Connected integrations (PostHog)</th>
            <th>API requests (PostHog)</th>
            <th>Tickets total (Pylon)</th>
            <th>Open tickets (Pylon)</th>
            <th>Open CS actions</th>
            <th>Last call (Fireflies)</th>
          </tr>
        </thead>
        <tbody id="crm-tbody"></tbody>
      </table>
    </div>

  </main>

  <script>
    // --- Constants ---
    const PLACEHOLDER_CELL = '<td class="crm-placeholder" title="Coming soon">—</td>';
    const PLACEHOLDER_COLS = 7; // columns 3–9

    // --- State ---
    let allRows  = [];
    let sortKey  = 'name';  // 'name' | 'contact'
    let sortDir  = 'asc';   // 'asc' | 'desc'
    let clickCounts = { name: 1, contact: 0 }; // name starts at 1 (already asc)

    // --- Helpers ---

    function escHtml(str) {
      return String(str ?? '')
        .replace(/&/g, '&amp;')
        .replace(/</g, '&lt;')
        .replace(/>/g, '&gt;')
        .replace(/"/g, '&quot;');
    }

    function isContactObj(c) {
      return c !== null && typeof c === 'object' && !Array.isArray(c);
    }

    function contactSortValue(company) {
      const c = isContactObj(company.contact) ? company.contact : null;
      if (!c || !c.name) return '\uffff'; // nulls sort last
      const parts = c.name.trim().split(/\s+/);
      const last  = parts.at(-1) ?? '';
      const first = parts.slice(0, -1).join(' ');
      return (last + ' ' + first).toLowerCase();
    }

    function renderRow(company) {
      const displayName = company.name ?? company.hubspot_id;
      const href        = escHtml(company.hubspot_url ?? '#');
      const nameCell    = `<td><a href="${href}" target="_blank" rel="noopener">${escHtml(displayName)}</a></td>`;

      const c = isContactObj(company.contact) ? company.contact : null;
      const contactCell = c
        ? `<td>
            <div class="crm-contact-name">${escHtml(c.name ?? '')}</div>
            <div class="crm-contact-email">${escHtml(c.email ?? '')}</div>
           </td>`
        : `<td class="crm-placeholder">—</td>`;

      return `<tr>${nameCell}${contactCell}${PLACEHOLDER_CELL.repeat(PLACEHOLDER_COLS)}</tr>`;
    }

    function searchMatch(company, query) {
      const q = query.toLowerCase();
      const c = isContactObj(company.contact) ? company.contact : null;
      return (
        (company.name   ?? '').toLowerCase().includes(q) ||
        (company.domain ?? '').toLowerCase().includes(q) ||
        (c?.name  ?? '').toLowerCase().includes(q) ||
        (c?.email ?? '').toLowerCase().includes(q)
      );
    }

    function sortedRows(rows) {
      return [...rows].sort((a, b) => {
        const dir = sortDir === 'asc' ? 1 : -1;
        if (sortKey === 'contact') {
          return dir * contactSortValue(a).localeCompare(contactSortValue(b));
        }
        return dir * (a.name ?? '').localeCompare(b.name ?? '');
      });
    }

    // --- Render ---

    function render() {
      const query    = document.getElementById('crm-search').value.trim();
      const filtered = query ? allRows.filter(c => searchMatch(c, query)) : allRows;
      const display  = sortedRows(filtered);

      document.getElementById('crm-tbody').innerHTML = display.map(renderRow).join('');

      const total = allRows.length;
      const shown = display.length;
      document.getElementById('crm-count').textContent =
        query ? `${shown} of ${total} customers` : `${total} customers`;
    }

    // --- Sort ---

    function setSort(key) {
      clickCounts[key] = (clickCounts[key] ?? 0) + 1;
      const click = clickCounts[key];

      if (click % 3 === 1) {
        sortKey = key; sortDir = 'asc';
      } else if (click % 3 === 2) {
        sortKey = key; sortDir = 'desc';
      } else {
        // Third click: reset to global default (name A→Z)
        sortKey = 'name'; sortDir = 'asc';
        clickCounts = { name: 1, contact: 0 };
      }

      // Update aria-sort on sortable headers
      const thName    = document.getElementById('th-name');
      const thContact = document.getElementById('th-contact');
      thName.removeAttribute('aria-sort');
      thContact.removeAttribute('aria-sort');
      const activeEl = sortKey === 'contact' ? thContact : thName;
      activeEl.setAttribute('aria-sort', sortDir === 'asc' ? 'ascending' : 'descending');

      render();
    }

    // --- Init ---

    fetch('./data/hubspot.json')
      .then(r => { if (!r.ok) throw new Error(r.status); return r.json(); })
      .then(data => {
        allRows = data;
        render();
      })
      .catch(() => {
        document.getElementById('crm-table-wrap').outerHTML =
          '<p class="crm-error">Customer data couldn\'t be loaded. Try refreshing the page.</p>';
      });

    document.getElementById('crm-search').addEventListener('input', render);
    document.getElementById('th-name').addEventListener('click', () => setSort('name'));
    document.getElementById('th-contact').addEventListener('click', () => setSort('contact'));
  </script>

</body>
</html>
```
