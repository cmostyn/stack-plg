# Connector Status Page Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a `connectors.html` page to the CS Analytics site showing live status, capabilities, and last-fetch stats for HubSpot, Pylon, PostHog, Fireflies, Google Drive, and Gmail.

**Architecture:** Each Node.js fetch script writes a `site/data/status-{connector}.json` file on success or failure. A new shared helper (`scripts/write-status.js`) handles the read-modify-write. `connectors.html` fetches all six status files at page load and renders a card grid. Existing scripts and their primary output files are unchanged — dashboard.html is not affected.

**Tech Stack:** Plain HTML/CSS/JS, Node.js fetch scripts, StackOne RPC API (`api.stackone.com/actions/rpc`), Cloudflare Pages (static hosting).

## Task 2: Shared status writer helper

**Files:**
- Create: `scripts/write-status.js`

- [ ] **Step 1: Create the helper**

```javascript
// scripts/write-status.js
'use strict';

const fs   = require('fs');
const path = require('path');

const STATUS_DIR = path.join(__dirname, '../site/data');

/**
 * Write a connector's status entry.
 * @param {string} connector  e.g. 'hubspot'
 * @param {'ok'|'error'|'not_configured'} status
 * @param {object} extra  Additional fields merged into the entry (e.g. { records: 55 })
 */
function writeStatus(connector, status, extra = {}) {
  const file = path.join(STATUS_DIR, `status-${connector}.json`);
  const entry = {
    status,
    last_fetch: new Date().toISOString(),
    ...extra,
  };
  fs.mkdirSync(STATUS_DIR, { recursive: true });
  fs.writeFileSync(file, JSON.stringify(entry, null, 2) + '\n');
  console.log(`[${connector}] Status written: ${status}`);
}

module.exports = { writeStatus };
```

## Task 4: New `fetch-fireflies.js`

**Files:**
- Create: `scripts/fetch-fireflies.js`

- [ ] **Step 1: Create the script**

```javascript
#!/usr/bin/env node
'use strict';

require('dotenv').config({ path: require('path').join(__dirname, '../.env') });

const { writeStatus } = require('./write-status');

const API_KEY    = process.env.STACKONE_API_KEY;
const ACCOUNT_ID = process.env.STACKONE_FIREFLIES_ACCOUNT_ID;

if (!API_KEY || !ACCOUNT_ID) {
  console.error('[fireflies] Missing STACKONE_API_KEY or STACKONE_FIREFLIES_ACCOUNT_ID');
  process.exit(1);
}

const AUTH = 'Basic ' + Buffer.from(API_KEY + ':').toString('base64');

async function rpc(action, body) {
  const res = await fetch('https://api.stackone.com/actions/rpc', {
    method: 'POST',
    headers: {
      'Authorization': AUTH,
      'x-account-id': ACCOUNT_ID,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({ action, body }),
  });
  if (!res.ok) throw new Error(`RPC ${action} failed: ${res.status} ${await res.text()}`);
  return res.json();
}

async function main() {
  const fromDate = new Date(Date.now() - 30 * 24 * 60 * 60 * 1000).toISOString();
  console.log(`[fireflies] Fetching transcripts from ${fromDate}...`);

  const data = await rpc('fireflies_list_transcripts', {
    body: { variables: { fromDate, limit: 50 } },
    query: {},
  });

  const transcripts = data?.data?.transcripts ?? data?.result?.transcripts ?? [];
  console.log(`[fireflies] Found ${transcripts.length} transcripts`);

  writeStatus('fireflies', 'ok', { records: transcripts.length });
}

main().catch(e => {
  console.error('[fireflies]', e.message);
  writeStatus('fireflies', 'error', { error: e.message });
  process.exit(1);
});
```

'use strict';

require('dotenv').config({ path: require('path').join(__dirname, '../.env') });

const { writeStatus } = require('./write-status');

const API_KEY    = process.env.STACKONE_API_KEY;
const ACCOUNT_ID = process.env.STACKONE_GOOGLEDRIVE_ACCOUNT_ID;

if (!API_KEY || !ACCOUNT_ID) {
  console.error('[googledrive] Missing STACKONE_API_KEY or STACKONE_GOOGLEDRIVE_ACCOUNT_ID');

const AUTH = 'Basic ' + Buffer.from(API_KEY + ':').toString('base64');

async function rpc(action, body) {

    method: 'POST',
    headers: {
      'Authorization': AUTH,
      'x-account-id': ACCOUNT_ID,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({ action, body }),
  });
  if (!res.ok) throw new Error(`RPC ${action} failed: ${res.status} ${await res.text()}`);
  return res.json();
}

async function main() {
  console.log('[googledrive] Fetching files...');

  const data = await rpc('googledrive_list_files', {
    query: {
      corpora: 'allDrives',
      includeItemsFromAllDrives: true,

      pageSize: 1000,
      fields: 'files(id)',

  console.log(`[googledrive] Found ${files.length} files`);

  writeStatus('googledrive', 'ok', { records: files.length });
}

main().catch(e => {

    function timeAgo(isoString) {
      if (!isoString) return null;
      const diff = Date.now() - new Date(isoString).getTime();
      const mins = Math.floor(diff / 60000);
      if (mins < 2) return 'just now';
      if (mins < 60) return `${mins} minutes ago`;
      const hrs = Math.floor(mins / 60);
      if (hrs < 24) return `${hrs === 1 ? '1 hour' : hrs + ' hours'} ago`;

        </div>
      `;

      grid.innerHTML = CONNECTORS.map((c, i) => renderCard(c, statuses[i])).join('');
    }
